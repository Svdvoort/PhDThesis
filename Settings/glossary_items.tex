% GLOSSARY ITEMS

\newglossaryentry{codeletion}{name=1p/19q co-deletion,
    description={A genetic mutation where the long arm of chromosome 1 and the short arm of chromosome 19 are absent. Glioma that are 1p/19q co-deleted show better survival than glioma that are 1p/19q intact}
}

\newglossaryentry{astro}{name=astrocytoma,
    description={Historically a type of glioma where the cancerous cells mutated from astrocyte cells. In the updated WHO guidelines it is a 1p/19q intact tumour, either IDH mutated or IDH wildtype},
    plural=astrocytomas
}

\newglossaryentry{gcell}{name=glial cell,
description={A type of brain cell that plays a supporting role in the central nervous system}}

\newglossaryentry{glioblastoma}{name=glioblastoma,
description={A grade IV glioma, the most aggresive type},
plural=glioblastomas}

\newglossaryentry{glioma}{name=glioma,
description={A type of brain tumour. The cancerous cells that make up a glioma mutated from (healthy) glial cells},
plural=gliomas}

\newglossaryentry{IDHmut}{name=IDH mutation,
first={\acrfull{IDH} mutation},
description={Genetic mutation of the IDH gene on chromosome 2 or chromosome 15. Glioma that are IDH mutated show a better survival than IDH wildtype glioma}
}

\newglossaryentry{MGMTmet}{name=MGMT methylation,
first={\acrfull{MGMT} methylation},
description={Genetic mutation of the MGMT gene on chromosome 10. Glioblastoma that are MGMT methylated show a better survival than MGMT unmethylated glioblastoma}
}


\newglossaryentry{oli}{name=oligodendroglioma,
    description={Historically a type of glioma where the cancerous cells mutated from oligodencrocyte cells. In the updated WHO guidelines it is the least aggresive glioma, and is 1p/19q co-deleted and IDH mutated},
    plural=oligodendrogliomas
}

\newglossaryentry{oliastro}{name=oligoastrocytoma,
    description={Historically a type of glioma where the cancerous cells mutated from a mixture of oligodencrocyte and astrocyte cells. It no longer exists in the updated WHO guidelines},
    plural=oligoastrocytomas
}


\newglossaryentry{tumor}{name=tumour,
    description={A lump of cancerous cells},
}
