\chapter{Publications}
% Nature style
\defbibenvironment{mypubs}
 {\list
     {}
     {\setlength{\leftmargin}{\bibhang}%
      % \setlength{\itemindent}{-\leftmargin}%
      \setlength{\itemsep}{\bibitemsep}%
      \setlength{\parsep}{.5cm}}}
  {\endlist}
  {\item}

\defbibfilter{conference_papers}{
    type=inproceedings and
    keyword=paper
}

\defbibfilter{conference_abstracts}{
    type=inproceedings and
    keyword=abstract
}

\defbibfilter{conference_presentations}{
    type=booklet and
    keyword=presentation
}

\defbibfilter{articles}{
    type=article or type=unpublished
}


\begin{refsection}[mypublications]
\nocite{*}
\newrefcontext[sorting=ydnt]
\printbibliography[omitnumbers=true,title={Journal Papers},env=mypubs,filter=articles,heading=subbibliography]
\newrefcontext[sorting=ydnt]
\printbibliography[omitnumbers=true,title={Book Chapters},env=mypubs,type=inbook,heading=subbibliography]
\newrefcontext[sorting=ydnt]
\printbibliography[omitnumbers=true,title={Conference Papers},env=mypubs,filter=conference_papers,heading=subbibliography]
\newrefcontext[sorting=ydnt]
\printbibliography[omitnumbers=true,title={Conference Abstracts},env=mypubs,filter=conference_abstracts,heading=subbibliography]
\newrefcontext[sorting=ydnt]
\printbibliography[omitnumbers=true,title={Presentations},env=mypubs,filter=conference_presentations,heading=subbibliography]

\vspace*{1cm}
\noindent* indicates equal contributions\\
$\dagger$ indicates presenting author

\end{refsection}
